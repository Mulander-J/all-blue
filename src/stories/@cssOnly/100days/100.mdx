$lineH: 120px;

	position: absolute;
	top: 50%;
	left: 50%;
	width: 400px;
	height: 400px;

	margin-left: -200px;

	overflow: hidden;
	background: #272c34;

.words {
	position: absolute;
	top: 35%;
	bottom: 35%;

	grid-column-gap: 25px;
}

	height: 100%;
}

	position: absolute;

	@for $i from 1 through 3 {
		&:nth-child(#{$i}) {
			animation-delay: 1s * ($i - 1);
		}
	}
}

$offset: $gap + $strokeW;

	top: 0;
	left: 0;
